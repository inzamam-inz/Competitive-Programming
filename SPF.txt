vector < ll > sie;
ll spf[ MAX + 7 ];

void sieve()
{
    for( ll i = 2; i < MAX; ++i ) {
        if (spf[ i ] == 0) {
            spf[ i ] = i;
            sie.push_back( i );
        }
        for ( ll j = 0; j < sie.size() && i * sie[ j ] <= MAX && sie[ j ] <= spf[ i ]; j++ )
            spf[ i * sie[ j ] ] = sie[ j ];
    }
}
