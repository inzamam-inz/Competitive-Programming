pair <Long, Long> fib (Long n) {
    if (n == 0) return {0, 1};

    auto p = fib(n >> 1);
    Long c = (((p.first * (2 * p.second - p.first)) % MOD) + MOD) % MOD;
    Long d = (((p.first * p.first + p.second * p.second) % MOD) + MOD) % MOD;
    
    if (n & 1) return {d, c + d};
    else       return {c, d};
}
